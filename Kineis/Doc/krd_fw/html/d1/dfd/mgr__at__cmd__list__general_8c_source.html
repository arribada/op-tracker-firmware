133 bool bMGR_AT_CMD_RCONF_cmd(uint8_t *pu8_cmdParamString __attribute__((unused)),
134  enum atcmd_type_t e_exec_mode)
135 {
136  enum KNS_status_t status;
137  struct KNS_CFG_radio_t radio_cfg;
138  uint8_t modulation[8];
139  uint16_t nbBits;
140 #if defined(KRD_FW_MP) || defined(KRD_FW_LP)
141  struct rfSettings_t hwSettings;
142 #endif
143 
144  if (e_exec_mode == ATCMD_STATUS_MODE) {
145  status = KNS_CFG_getRadioInfo(&radio_cfg);
146  if (status != KNS_STATUS_OK)
147  return bMGR_AT_CMD_logFailedMsg((enum ERROR_RETURN_T) status);
148  if (radio_cfg.modulation == KNS_TX_MOD_LDA2)
149  strcpy((char*)modulation, "LDA2");
150  else if (radio_cfg.modulation == KNS_TX_MOD_LDA2L)
151  strcpy((char*)modulation, "LDA2L");
152  else if (radio_cfg.modulation == KNS_TX_MOD_VLDA4)
153  strcpy((char*)modulation, "VLDA4");
154  else if (radio_cfg.modulation == KNS_TX_MOD_HDA4)
155  strcpy((char*)modulation, "HDA4");
156  else if (radio_cfg.modulation == KNS_TX_MOD_LDK)
157  strcpy((char*)modulation, "LDK");
158  else
159  strcpy((char*)modulation, "UNKNOWN");
160  MCU_AT_CONSOLE_send("+RCONF=%u,%u,%d,%s\r\n", radio_cfg.min_frequency,
161  radio_cfg.max_frequency, radio_cfg.rf_level, modulation);
162 
163  /** @todo PRODEV-68: check radio conf versus HW settings for other platforms */
164 #if defined(KRD_FW_MP) || defined(KRD_FW_LP)
165  if (MCU_MISC_getSettingsHwRf(radio_cfg.rf_level, &hwSettings) == KNS_STATUS_OK)
166  return bMGR_AT_CMD_logSucceedMsg();
167  return bMGR_AT_CMD_logFailedMsg(ERROR_INCOMPATIBLE_VALUE);
168 #else
169  return bMGR_AT_CMD_logSucceedMsg();
170 #endif
171  }
172  if (e_exec_mode == ATCMD_ACTION_MODE) {
173  while(pu8_cmdParamString[strlen((char*)pu8_cmdParamString) - 1] == '\r' ||
174  pu8_cmdParamString[strlen((char*)pu8_cmdParamString) - 1] == '\n')
175  pu8_cmdParamString[strlen((char*)pu8_cmdParamString) - 1] = '\0';
176  nbBits = u16MGR_AT_CMD_convertAsciiBinary(pu8_cmdParamString + 9,
177  strlen((char*)(pu8_cmdParamString + 9)));
178 
179  if (nbBits != 128)
180  return bMGR_AT_CMD_logFailedMsg((enum ERROR_RETURN_T) KNS_STATUS_RADIO_CONF_BAD);
181 
182  status = KNS_CFG_setRadioInfo(pu8_cmdParamString + 9);
183  if (status != KNS_STATUS_OK)
184  return bMGR_AT_CMD_logFailedMsg((enum ERROR_RETURN_T) status);
185  return bMGR_AT_CMD_logSucceedMsg();
186  }
187  return bMGR_AT_CMD_logFailedMsg(ERROR_UNKNOWN_AT_CMD);
188 }

203 bool bMGR_AT_CMD_LPM_cmd(uint8_t *pu8_cmdParamString __attribute__((unused)),
204  enum atcmd_type_t e_exec_mode)
205 {
206  int16_t scanParamRes;
207  uint16_t lpm; /** @todo PRODEV-88:need to keep uint16_t remove compilation error.
208  * leads to hardfault error if using uint8_t
209  */
210 
211  if (e_exec_mode == ATCMD_STATUS_MODE) {
212  MCU_AT_CONSOLE_send("+LPM=0x%X\r\n", lpm_config.allowedLPMbitmap);
213  return bMGR_AT_CMD_logSucceedMsg();
214  }
215  if (e_exec_mode == ATCMD_ACTION_MODE) {
216 
217  /** Extract USER DATA from pu8_cmdParamString */
218  scanParamRes = sscanf((const char *)pu8_cmdParamString, (const char *)"AT+LPM=0x%hX", &lpm);
219  if (scanParamRes == 1) {
220  lpm &= LOW_POWER_MODE_NONE | LOW_POWER_MODE_SLEEP | LOW_POWER_MODE_STOP |
221  LOW_POWER_MODE_STANDBY | LOW_POWER_MODE_SHUTDOWN;
222  lpm_config.allowedLPMbitmap = (uint8_t)lpm;
223  bMGR_AT_CMD_logSucceedMsg();
224  }
225  if (scanParamRes != 1)
226  return bMGR_AT_CMD_logFailedMsg(ERROR_PARAMETER_FORMAT);
227  }
228 
229  return false;
230 }
