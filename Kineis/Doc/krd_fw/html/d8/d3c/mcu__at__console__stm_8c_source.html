72 /** @brief RX interrupt handler for 7 or 8 bits data word length .
73  *
74  * This function is highly based on STM32HAL_UART expect callback is called at each received
75  * character. This allow to treat RX received characters as a continuous stream.
76  *
77  * At each received character, the callback is said to consume data, it updates the buffer pointer
78  * and the number-of-valid-char reduced by what was consummed.
79  *
80  * @note In case of overflow on RX buffer, force write from the beginning of the buffer pRxBuffPtr.
81  * thus, the entire buffer may be lost. Actually, such case may only happen in case:
82  * * user sent too much data in console, this is not an expected scenario
83  * * user has sent some garbage before its several AT cmds, leading accumuation to overflow.
84  * This is not an expected behaviour as well.
85  *
86  * @todo circular buffer may be implemented instead of overflowing sometimes.
87  *
88  * STM32HAL_UART base function prototype:
89  @verbatim
90 static void UART_RxISR_8BIT(UART_HandleTypeDef *huart)
91  @endverbatim
92  *
93  * @param huart UART handle.
94  * @retval None

104  pu8_RxBuffer = huart->pRxBuffPtr - i16_nbRxValidChar;
105  //< pu8_RxBuffer points to the 1st element of the RX array
106 
107  /* Check that a Rx process is ongoing */
108  if (huart->RxState == HAL_UART_STATE_BUSY_RX) {
109  /* consume all incomming data in same interrupt. RXNE is set by HW and cleared by
110  * reading RDR reg
111  */
112  while ((READ_REG(huart->Instance->ISR) & ‪USART_ISR_RXNE) != 0U) {
113  uhdata = (uint16_t) READ_REG(huart->Instance->RDR);

120  huart->RxXferCount;

122  } else {
123  huart->pRxBuffPtr++;

150 /** @brief Enable and start RX interrupt from UART
151  *
152  * This function is highly based on STM32HAL_UART expect it called internal interrupt handler
153  * instead of generic one.
154  *
155  @verbatim
156 HAL_StatusTypeDef UART_Start_Receive_IT(UART_HandleTypeDef *huart, uint8_t *pData, uint16_t Size);
157 HAL_StatusTypeDef HAL_UART_Receive_IT(UART_HandleTypeDef *huart, uint8_t *pData, uint16_t Size);
158  @endverbatim
159  * @param huart UART handle.
160  * @param pData Pointer to data buffer (u8 or u16 data elements).
161  * @param Size Amount of data elements (u8 or u16) to be received.
162  * @retval HAL status
163  */
164 static HAL_StatusTypeDef ‪KINEIS_UART_StartRx_IT(UART_HandleTypeDef *huart, uint8_t *pData,
165  uint16_t Size)
166 {
167  if (huart == NULL)
168  return HAL_ERROR;
169 
170  /* Check that a Rx process is not already ongoing */
171  if (huart->RxState == HAL_UART_STATE_READY) {
172  if ((pData == NULL) || (Size == 0U))
173  return HAL_ERROR;
174 
175  __HAL_LOCK(huart);
176 
177  huart->pRxBuffPtr = pData;
178  huart->RxXferSize = Size;
179  huart->RxXferCount = Size;
180  huart->RxISR = NULL;
181 
182  /* Computation of UART mask to apply to RDR register */
183  UART_MASK_COMPUTATION(huart);
184 
185  huart->ErrorCode = HAL_UART_ERROR_NONE;
186  huart->RxState = HAL_UART_STATE_BUSY_RX;
187 
188  /* Enable the UART Error Interrupt: (Frame error, noise error, overrun error) */
189  SET_BIT(huart->Instance->CR3, USART_CR3_EIE);
190 
191  /* Set the Rx ISR function pointer according to the data word length */
192  if ((huart->Init.WordLength != UART_WORDLENGTH_9B) ||
193  (huart->Init.Parity != UART_PARITY_NONE)) {
194  huart->RxISR = ‪KINEIS_RxISR_8BIT;
195  }
196 
197  __HAL_UNLOCK(huart);
198 
199  /* Enable the UART Parity Error interrupt and Data Register Not Empty interrupt */
200 #if defined(STM32WLE5xx) || defined(STM32WL55xx)
201  ATOMIC_SET_BIT(huart->Instance->CR1, USART_CR1_PEIE | USART_CR1_RXNEIE_RXFNEIE);
202 #else
203  SET_BIT(huart->Instance->CR1, USART_CR1_PEIE | USART_CR1_RXNEIE);
204 #endif

206  } else {
207  return HAL_BUSY;
208  }
