217 uint16_t ‪u16MGR_AT_CMD_convertAsciiBinary(uint8_t *pu8InputBuffer, uint16_t u16_charNb)
218 {
219  uint16_t u16_index = 0;
220  uint8_t u8_high, u8_low;
221 
222  for (u16_index = 0; u16_index < (u16_charNb / 2); u16_index++) {
223  u8_high = ‪u8UTIL_convertCharToHex4bits(pu8InputBuffer[2 * u16_index]);
224  u8_low = ‪u8UTIL_convertCharToHex4bits(pu8InputBuffer[2 * u16_index + 1]);
225  if ((u8_high != 0xFF) && (u8_low != 0xFF))
226  pu8InputBuffer[u16_index] = (u8_high << 4) | u8_low;
227  else
228  return 0;
229  }
230 
231  /** Case : characters number is an odd number */
232  if (u16_charNb % 2 != 0) {
233  u8_high = ‪u8UTIL_convertCharToHex4bits(pu8InputBuffer[u16_charNb - 1]);
234  u8_low = 0;
235  if (u8_high != 0xFF)
236  pu8InputBuffer[u16_index++] = (u8_high << 4) | u8_low;
237  else
238  return 0;
239  }
240 
241  /** Set other bytes to zero */
242  for (; u16_index < u16_charNb; u16_index++)
243  pu8InputBuffer[u16_index] = 0;
244 
245  /** Return data length in bits */
246  return (((u16_charNb / 2) * 8) + ((u16_charNb % 2) ? 4 : 0));
247 }
