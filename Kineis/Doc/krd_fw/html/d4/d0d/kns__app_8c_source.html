153 void ‪KNS_APP_stdln_loop(void)
154 {
155  /** Static variables for state machine.
156  * @note The implementation as a state machine is mainly usefull with kineis baremetal OS.
157  * The standalone application should be much more simple with a real OS
158  */
159  static
160  ‪__attribute__((__section__(".retentionRamData")))
161  uint8_t state;
162 
163  /** Static variable used with rand_r() function to generate random USER DATA in messages
164  * sent to Kineis stack.
165  * This will work also with STANDBY LPM as the seed tis mapped in SRAM2 memory here
166  * @note using rand() instead of rand_r() may lead to generate several messages with the
167  * same USER DATA payload
168  */
169  static
170  ‪__attribute__((__section__(".retentionRamData")))
171  uint32_t seed;
172 
173  uint16_t ‪idx;
174  struct ‪KNS_MAC_appEvt_t appEvt;
175  uint8_t buffer_tx[‪KNS_MAC_USRDATA_MAXLEN];
176 
177  switch (state) {
178  case 0: { /** Init MAC profile */
179  ‪KNS_APP_stdln_startMacPrfl();
180  state++;
181  return;
182  }
183  case 1: { /** Wait for Kineis stack replying:
184  * * OK to Init
185  */
186  enum ‪KNS_status_t status = ‪KNS_STATUS_OK;
187  struct ‪KNS_MAC_srvcEvt_t srvcEvt;
188 
189  for (‪status = ‪KNS_Q_pop(‪KNS_Q_UL_MAC2APP, (void *)&srvcEvt);
190  ‪status != ‪KNS_STATUS_QEMPTY;
191  ‪status = ‪KNS_Q_pop(‪KNS_Q_UL_MAC2APP, (void *)&srvcEvt)) {
192  if (‪status == ‪KNS_STATUS_OK) {
193  switch (srvcEvt.id) {
194  case (‪KNS_MAC_OK):
195  ‪MGR_LOG_DEBUG("[%s] MAC profile init OK\r\n", __func__);
196  state++;
197  break;
198  case (‪KNS_MAC_ERROR):
199  ‪MGR_LOG_DEBUG("[%s] MAC profile error\r\n", __func__);
200  ‪TEST_FAIL();
201  break;
202  default:
203  ‪TEST_FAIL();
204  break;
205  }
206  }
207  }
208  return;
209  }
210  case 2: { /** Send data event */
211  struct ‪KNS_CFG_radio_t device_radio_cfg;
212 
213  /** Initialize buffer with random data */
214  for (‪idx = 0; ‪idx < sizeof(buffer_tx); ‪idx++)
215  buffer_tx[‪idx] = rand_r(((unsigned int *)&seed));
216 
217  /** ---- OPTIONAL BLOCK ---- START -- needed if radio conf not hardcoded ----- */
218  ‪kns_assert(‪KNS_CFG_getRadioInfo(&device_radio_cfg) == ‪KNS_STATUS_OK);
219 
220  appEvt.‪id = ‪KNS_MAC_SEND_DATA;
221  for (‪idx = 0; ‪idx < sizeof(buffer_tx); ‪idx++)
222  appEvt.‪data_ctxt.‪usrdata[‪idx] = buffer_tx[‪idx];
223  appEvt.‪data_ctxt.‪sf = ‪KNS_SF_NO_SERVICE;
224 
225  switch (device_radio_cfg.‪modulation) {
226  case (‪KNS_TX_MOD_LDA2):
227  /** max payload size in this example, cf spec for smaller packets */
228  appEvt.‪data_ctxt.‪usrdata_bitlen = 192;
229  break;
230  case (‪KNS_TX_MOD_LDA2L):
231  appEvt.‪data_ctxt.‪usrdata_bitlen = 196;
232  break;
233  case (‪KNS_TX_MOD_VLDA4):
234  appEvt.‪data_ctxt.‪usrdata_bitlen = 24;
235  break;
236  case (‪KNS_TX_MOD_LDK):
237  appEvt.‪data_ctxt.‪usrdata_bitlen = 152;
238  break;
239  default:
240  ‪kns_assert(0); // wrong modulation flashed into device
241  break;
242  }
243  /** ---- OPTIONAL BLOCK ---- END --------------------------------------------- */
244 
245  ‪MGR_LOG_DEBUG("[%s] request to send 0x", __func__);
246  ‪MGR_LOG_array(appEvt.‪data_ctxt.‪usrdata, (appEvt.‪data_ctxt.‪usrdata_bitlen+7)>>3);
247 
248  ‪TEST_ASSERT(‪KNS_Q_push(‪KNS_Q_DL_APP2MAC, (void *)&appEvt) == ‪KNS_STATUS_OK);
249 
250  state++; /* go to next state, and return to let higher task to process event */
251  return;
252  }
253  case 3: { /** Wait for Kineis stack replying:
254  * * OK to send frame
255  * * TX done for previous transmit
256  */
257  enum ‪KNS_status_t status = ‪KNS_STATUS_OK;
258  struct ‪KNS_MAC_srvcEvt_t srvcEvt;
259 
260  for (‪status = ‪KNS_Q_pop(‪KNS_Q_UL_MAC2APP, (void *)&srvcEvt);
261  ‪status != ‪KNS_STATUS_QEMPTY;
262  ‪status = ‪KNS_Q_pop(‪KNS_Q_UL_MAC2APP, (void *)&srvcEvt)) {
263  if (‪status == ‪KNS_STATUS_OK) {
264  switch (srvcEvt.id) {
265  case (‪KNS_MAC_OK):
266  if (srvcEvt.app_evt == ‪KNS_MAC_SEND_DATA) {
267  ‪MGR_LOG_DEBUG("[%s] OK to send 0x", __func__);
268  ‪MGR_LOG_array(srvcEvt.tx_ctxt.data,
269 // (srvcEvt.tx_ctxt.data_bitlen+7)>>3);
270  4); // limit to 4 bytes for real-time
271  }
272  break;
273  case (‪KNS_MAC_TX_DONE):
274  ‪MGR_LOG_DEBUG("[%s] TX done for 0x", __func__);
275  ‪MGR_LOG_array(srvcEvt.tx_ctxt.data,
276  (srvcEvt.tx_ctxt.data_bitlen+7)>>3);
277  ‪TEST_PASS();
278  /** Exit MAC as test succedded. It should reply OK at next
279  * step
280  */
281  ‪KNS_APP_stdln_stopMacPrfl();
282  state++;
283  break;
284  case (‪KNS_MAC_TX_TIMEOUT):
285  ‪MGR_LOG_DEBUG("[%s] TX timeout for 0x", __func__);
286  ‪MGR_LOG_array(srvcEvt.tx_ctxt.data,
287  (srvcEvt.tx_ctxt.data_bitlen+7)>>3);
288  ‪TEST_FAIL();
289  state++;
290  break;
291  case (‪KNS_MAC_ERROR):
292  /** MAC ERROR can occure when FIFO is full, as expected per
293  * this application
294  */
295  if (srvcEvt.app_evt == ‪KNS_MAC_SEND_DATA) {
296  ‪MGR_LOG_DEBUG("[%s] cannot send 0x", __func__);
297  ‪MGR_LOG_array(srvcEvt.tx_ctxt.data,
298 // (srvcEvt.tx_ctxt.data_bitlen+7)>>3);
299  4); // limit to 4 bytes for real-time
300  }
301  ‪TEST_FAIL();
302  state++;
303  break;
304  default:
305  ‪TEST_FAIL();
306  break;
307  }
308  }
309  }
310  return;
311  }
312  case 4: { /** Wait for Kineis stack replying:
313  * * OK to Init
314  */
315  enum ‪KNS_status_t ‪status = ‪KNS_STATUS_OK;
316  struct ‪KNS_MAC_srvcEvt_t srvcEvt;
317 
318  for (‪status = ‪KNS_Q_pop(‪KNS_Q_UL_MAC2APP, (void *)&srvcEvt);
319  ‪status != ‪KNS_STATUS_QEMPTY;
320  ‪status = ‪KNS_Q_pop(‪KNS_Q_UL_MAC2APP, (void *)&srvcEvt)) {
321  if (‪status == ‪KNS_STATUS_OK) {
322  switch (srvcEvt.id) {
323  case (‪KNS_MAC_OK):
324  ‪MGR_LOG_DEBUG("[%s] OK from MAC\r\n", __func__);
325  state++;
326  break;
327  case (‪KNS_MAC_ERROR):
328  ‪MGR_LOG_DEBUG("[%s] MAC ERROR\r\n", __func__);
329  ‪TEST_FAIL();
330  state++;
331  break;
332  default:
333  ‪TEST_FAIL();
334  break;
335  }
336  }
337  }
338  return;
339  }
340  case 5: { /** Wait for Kineis stack replying:*/
341  break;
342  }
343  default:
344  ‪kns_assert(0);
345  break;
346  }
