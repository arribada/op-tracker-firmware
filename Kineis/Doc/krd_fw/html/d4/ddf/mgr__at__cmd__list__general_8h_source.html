138 /** @brief Set/Get low power mode "AT+LPM"
139  *
140  * So far, on all kineis platforms and reference designs, LPm is described as \ref MgrLpm_LPM_t:
141  * LOW_POWER_MODE_NONE = 0x00,
142  * LOW_POWER_MODE_SLEEP = 0x01,
143  * LOW_POWER_MODE_STOP = 0x02,
144  * LOW_POWER_MODE_STANDBY = 0x04,
145  * LOW_POWER_MODE_SHUTDOWN = 0x08
146  *
147  * 1) "AT+LPM=<LPM capabilities bitmap" will configure the expected LPM.
148  * the FW will then allow to reach the deepest LPM configured inthis bitmap.
149  *
150  * @attention Ensure your configuration fits the LPM capabilities implemented in the FW
151  *
152  * 2) "AT+LPM=?" will reply te LPM configuration bitmap
153  *
154  * Response format: "+LPM=<bitmap>" depending on your implementation.
155  *
156  * @param[in] pu8_cmdParamString: string containing AT command
157  * @param[in] e_exec_mode: type of the command (status command or action command)
158  *
159  * @return true if command is correctly received and processed, false if error
160  */
161 bool bMGR_AT_CMD_LPM_cmd(uint8_t *pu8_cmdParamString, enum atcmd_type_t e_exec_mode);

171  * represents the device message counter used for antireplay and authentication of the radio frame.
172  *
173  * @param[in] pu8_cmdParamString: string containing AT command
