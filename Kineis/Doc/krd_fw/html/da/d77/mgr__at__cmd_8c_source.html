28 /** Define AT commands FIFO length. As long as sending AT+TX one by one, it is not that usefull to
29  * enqueue AT commands so far. Later, this may be useful to increase FIFO in case application or
30  * GUI is sending several commands in a raw.
31  *
32  * @attention AT CMD FIFO_MAX_SIZE must be superior or equal to 2
33  *
34  * @attention Ensure there is always one extra free space in fifo between write index and read
35  * index meaning that if you expect to store 3 AT cmd at maximum, FIFO size should be
36  * 3+1 minimum if you want to avoid overflow.
37  */
38 #define FIFO_MAX_SIZE 4
39 #ifdef USE_HDA4
40 #define FRAME_MAX_LEN 1280
41 #else
42 #define FRAME_MAX_LEN 128
43 #endif

46 #error "FIFO_MAX_SIZE must be superior or equal to 2"

67 /**
68  * @brief API used to extract the latest AT cmds from the incoming received data stream.
69  *
70  * This fct is looking for the start pattern ("AT+") and the end pattern ("\r\n") in the stream.
71  * If several frames are received in a row, only the last AT cmd is extracted
72  *
73  * Once AT cmd is found, it is removed from the stream and stored into the FIFO if not full.
74  *
75  * As this fct is the entry point into FW, it is real-time critical and may be more robust. Some
76  * design limitations are describerd below.
77  *
78  * @attention This fct may be called from ISR context
79  *
80  * @note Any garbage after the found AT cmd is also removed from stream. But the garbage before
81  * remains in the original stream.
82  *
83  * @param[in,out] pu8_RxBuffer pointer to start of RX buffer
84  * @param[in,out] pi16_nbRxValidChar number of valid charecters in RX buffer
85  *
86  * @retval true if a valid frame was extracted, false if nothing was extracted
87  */
88 static bool MGR_AT_CMD_parseStreamCb(uint8_t *pu8_RxBuffer, int16_t *pi16_nbRxValidChar)
89 {
90  int16_t idxEnd = 0;
91  int16_t idxStart = 0;
92  int16_t i16_atcmdLen = 0;
93  bool isEOLdetected = false;
94  bool isFirstCharDetected = false;
95 
96  char C = (char)(pu8_RxBuffer[*pi16_nbRxValidChar - 1]);
97  /* Process buffer only once termination caracter has been found */
98  if (C != '\r')
99  return false;
100 
101  /* AT cmd starts with 'AT+' and ends with '\r\n'. Minimum packet length is ("AT+\r\n") */
102  if (*pi16_nbRxValidChar < 5)
103  return false;
104 
105  /* Search beginning of AT cmd from the end of the buffer */
106  for (idxStart = *pi16_nbRxValidChar - 1; idxStart >= 0; idxStart--)
107  if (pu8_RxBuffer[idxStart] == 'A' &&
108  pu8_RxBuffer[idxStart + 1] == 'T' &&
109  pu8_RxBuffer[idxStart + 2] == '+') {
110  isFirstCharDetected = true;
111  break;
112  }
113 
114  /* Return if we can not find the beginning of the frame*/
115  if (!isFirstCharDetected)
116  return false;
117 
118  /* Find last characters of the frame : '\r\n' */
119  for (idxEnd = idxStart + 2; idxEnd <= *pi16_nbRxValidChar; idxEnd++) {
120  if ((pu8_RxBuffer[idxEnd - 2] == '\r' && pu8_RxBuffer[idxEnd - 1] == '\n') ||
121  (pu8_RxBuffer[idxEnd - 1] == '\r')){
122  isEOLdetected = true;
123  break;
124  }
125  }
126  /* Return if we can not find the EOL characters */
127  if (!isEOLdetected)
128  return false;
129 
130  /* As some AT cmd was consummed from the end at idxStart position,
131  * idxStart characters remaing to be parsed at the beginning of the buffer.
132  * Assume they are all valid.
133  *
134  */
135  *pi16_nbRxValidChar = idxStart;
136 
137  /* In case FIFO is full (read index is reached), skip this new AT CMD. We need to wait for
138  * FW to consume the previous AT CMDs before.
139  */
140  if ( ((s_atcmdfifo.u8_widx+1) % FIFO_MAX_SIZE) == (s_atcmdfifo.u8_ridx % FIFO_MAX_SIZE))
141  return true;
142  /* Set the frame in the UART fifo */
143  i16_atcmdLen = idxEnd - idxStart;
144  /* Check AT cmd length overflow. Limit AT cmd len to maximum if overflow was
145  * detected (reserve last character for end-of-string'\0').
146  */
147  if ((i16_atcmdLen + 1) > FRAME_MAX_LEN)
148  i16_atcmdLen = FRAME_MAX_LEN - 1;
149  memcpy(s_atcmdfifo.au8_fifo[s_atcmdfifo.u8_widx % FIFO_MAX_SIZE],
150  &pu8_RxBuffer[idxStart], i16_atcmdLen);
151  s_atcmdfifo.au8_fifo[s_atcmdfifo.u8_widx % FIFO_MAX_SIZE][i16_atcmdLen] = '\0';
152  /* Increment write index to next free position */
153  s_atcmdfifo.u8_widx++;
154 
155  /* In case read index is reached, skip oldest stored AT cmd to let a newest one to come */
156  if (s_atcmdfifo.u8_widx % FIFO_MAX_SIZE == s_atcmdfifo.u8_ridx % FIFO_MAX_SIZE)
157  kns_assert(0); // should no more happen
158  // s_atcmdfifo.u8_ridx++;
159 
160  return true;
161 }

163 /**
164  * @brief Identifies AT command and its type (AT+XYZ=... or AT+XYZ=? or AT+XYZ? mode)
165  *
166  * AT cmds with "=?\r\n" (e.g. "AT+ID=?\r\n") is a status-type command. It is used to get some
167  * information from the device.
168  *
169  * AT cmds without pattern above is an action-type command. It is used to set some information into
170  * the device (e.g. "AT+TX=<...>\r\n").
171  *
172  * @note This fct may be called from ISR context
173  *
174  * @param[in] pu8_atcmd pointer to the AT command
175  *
176  * @returns AT cmd info (e.g. index and execution type)
177  */
178 static struct atcmd_info_t MGR_AT_CMD_getAtType(uint8_t *pu8_atcmd)
179 {
180  uint8_t __u8_k;
181  uint8_t *p_atcmd_start;
182  uint8_t u8foundCmdIndex = ATCMD_UNKNOWN_COMMAND;
183  uint16_t u16_offset_search = 0;
184  struct atcmd_info_t atcmd_info;
185 
186 
187  atcmd_info.ATcmdIndex = u8foundCmdIndex;
188 
189  /* Loop over table and check for match */
190  for (__u8_k = 0; __u8_k < ATCMD_MAX_COUNT; __u8_k++) {
191  u16_offset_search = cas_atcmd_list_array[__u8_k].u8_cmdNameLen;
192 
193  if (bUTIL_strcmp((uint8_t *)cas_atcmd_list_array[__u8_k].pu8_cmdNameString,
194  pu8_atcmd,
195  u16_offset_search)) {
196  u8foundCmdIndex = (enum atcmd_idx_t)__u8_k;
197  break;
198  }
199  }
200  /* If command is found in list, check type */
201  if (u8foundCmdIndex < ATCMD_UNKNOWN_COMMAND) {
202  u16_offset_search = cas_atcmd_list_array[u8foundCmdIndex].u8_cmdNameLen;
203  p_atcmd_start = pu8_atcmd;
204 
205  /* Checks if next characters after command are either "=?\r", "=?\n" or
206  * "=<params>\r" or "=<params>\n"
207  *
208  * @note "AT+<cmd>\r\n" is considered as action mode. it will be filtered in the
209  * next parsing level.
210  */
211  if ((p_atcmd_start[u16_offset_search] == '=')
212  && (p_atcmd_start[u16_offset_search+1] == '?')
213  && ((p_atcmd_start[u16_offset_search+2] == '\r')
214  || (p_atcmd_start[u16_offset_search+2] == '\n'))) {
215  atcmd_info.ATcmdExecType = ATCMD_STATUS_MODE;
216  atcmd_info.ATcmdIndex = u8foundCmdIndex;
217  } else if ((p_atcmd_start[u16_offset_search] == '=')
218  || (p_atcmd_start[u16_offset_search] == '\r')
219  || (p_atcmd_start[u16_offset_search] == '\n')) {
220  atcmd_info.ATcmdExecType = ATCMD_ACTION_MODE;
221  atcmd_info.ATcmdIndex = u8foundCmdIndex;
222  } else
223  atcmd_info.ATcmdIndex = ATCMD_UNKNOWN_COMMAND;
224  }
225  return atcmd_info;
226 }

249 bool MGR_AT_CMD_decodeAt(uint8_t *pu8_atcmd)
250 {
251  bool status = false;
252  struct atcmd_info_t atcmdInfo;
253 
254  if (pu8_atcmd != NULL) {
255  atcmdInfo = MGR_AT_CMD_getAtType(pu8_atcmd);
256 
257  if ((atcmdInfo.ATcmdIndex < ATCMD_UNKNOWN_COMMAND) &&
258  (cas_atcmd_list_array[atcmdInfo.ATcmdIndex].f_ht_cmd_fun_proc != NULL))
259  status = (cas_atcmd_list_array[atcmdInfo.ATcmdIndex].f_ht_cmd_fun_proc)(
260  pu8_atcmd,
261  atcmdInfo.ATcmdExecType);
262  else {
263  bMGR_AT_CMD_logFailedMsg(ERROR_UNKNOWN_AT_CMD);
264  MGR_LOG_VERBOSE("[ERROR] Unknown AT command\r\n");
265  }
266  } else {
267  bMGR_AT_CMD_logFailedMsg(ERROR_UNKNOWN);
268  MGR_LOG_DEBUG("[ERROR] input param is Nil\r\n");
269  }
270  return status;
271 }
