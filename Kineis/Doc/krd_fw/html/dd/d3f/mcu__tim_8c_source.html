146 enum mcu_tim_status_t MCU_TIM_start(enum mcu_tim_hdlr hdlr, uint32_t timeout_ms)
147 {
148  TIM_HandleTypeDef *htim = &htim16;
149  RTC_HandleTypeDef *hrtc_local;
150  uint32_t cnt_val, cnt_val_max;
151 
152  MGR_LOG_VERBOSE("%d: %s %d\r\n", hdlr, __FUNCTION__, __LINE__);
153 
154  /** Configure timer handler */
155  switch (hdlr) {
156  case MCU_TIM_HDLR_TX_TIMEOUT:
157  htim = &htim16;
158  break;
159  case MCU_TIM_HDLR_TX_PERIOD:
160  hrtc_local = &hrtc;
161  break;
162  default:
163  return MCU_TIM_STATUS_ERROR;
164  break;
165  }
166 
167  /** Configure timer settings */
168  switch (hdlr) {
169  case MCU_TIM_HDLR_TX_TIMEOUT:
170  /** As htim16 ARR are 16 long, check delay is not too big.
171  */
172  cnt_val = timeout_ms * 2 - 1;
173  cnt_val_max = (1 << 16) - 1;
174  if (cnt_val > cnt_val_max)
175  return MCU_TIM_STATUS_ERROR;
176  MGR_LOG_VERBOSE("start timer %d for %d ms, cnt=%d, cnt_max=%d\r\n",
177  hdlr, timeout_ms, cnt_val, cnt_val_max);
178 
179  __HAL_TIM_CLEAR_FLAG(htim, TIM_IT_UPDATE);
180  __HAL_TIM_SET_COUNTER(htim, 0);
181  __HAL_TIM_SET_AUTORELOAD(htim, cnt_val);
182  HAL_TIM_Base_Start_IT(htim);
183  break;
184  case MCU_TIM_HDLR_TX_PERIOD:
185  cnt_val = timeout_ms / 1000;
186  cnt_val--; /** With this config of RTC timer (1s tick), need to reduce count down
187  * by one
188  */
189  cnt_val_max = (1 << 16) - 1;
190  if (cnt_val > cnt_val_max)
191  return MCU_TIM_STATUS_ERROR;
192  MGR_LOG_VERBOSE("start timer %d for %d ms, cnt=%d, cnt_max=%d\r\n",
193  hdlr, timeout_ms, cnt_val, cnt_val_max);
194  if (HAL_RTCEx_SetWakeUpTimer_IT(hrtc_local, cnt_val,
195  RTC_WAKEUPCLOCK_CK_SPRE_16BITS, 0) != HAL_OK)
196  Error_Handler();
197  break;
198  default:
199  return MCU_TIM_STATUS_ERROR;
200  break;
201  }
202 
203  return MCU_TIM_STATUS_OK;
204 }

206 enum mcu_tim_status_t MCU_TIM_getCount(enum mcu_tim_hdlr hdlr, uint32_t *elapsed_time_ms)
207 {
208  TIM_HandleTypeDef *htim;
209  RTC_HandleTypeDef *hrtc_local;
210 
211  switch (hdlr) {
212  case MCU_TIM_HDLR_TX_TIMEOUT:
213  htim = &htim16;
214  /** Get counter value and divide it by 2 as it is currently a 500us-step counter */
215  *elapsed_time_ms = __HAL_TIM_GET_COUNTER(htim) / 2;
216  break;
217  case MCU_TIM_HDLR_TX_PERIOD:
218  hrtc_local = &hrtc;
219  /** Get counter value and divide it by 2 as it is currently a 500us-step counter */
220  *elapsed_time_ms = 1000 * HAL_RTCEx_GetWakeUpTimer(hrtc_local);
221  break;
222  default:
223  return MCU_TIM_STATUS_ERROR;
224  break;
225  }
226 
227  MGR_LOG_VERBOSE("%d: %s %d: %d ms\r\n", hdlr, __FUNCTION__, __LINE__, *elapsed_time_ms);
228 
229  return MCU_TIM_STATUS_OK;
230 }
