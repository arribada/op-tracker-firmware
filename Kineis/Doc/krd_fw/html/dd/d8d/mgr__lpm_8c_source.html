99 enum KNS_status_t MGR_LPM_enter(struct MgrLpm_EnvConfig_t env_config,
100  struct MgrLpm_ctxt_t *mgr_lpm_ctxt)
101 {
102  enum MgrLpm_LPM_t deepest_lpm;
103 
104  //> Configuration : allowedLPMbitmap should not be equal to 0. It means no LPM is enable.
105  if (env_config.allowedLPMbitmap == 0)
106  return KNS_STATUS_ERROR;
107 
108  //> Get the deepest LPM which clients can accept
109  deepest_lpm = eMGR_LPM_clientRequest();
110 
111  //> Check if the deepest LPM exists
112  while ((deepest_lpm & env_config.allowedLPMbitmap) == 0) {
113  deepest_lpm = deepest_lpm >> 1;
114  if (deepest_lpm == 0)
115  break;
116  }
117 
118  //> Notify each client with the deepest chosen LPM
119  vMGR_LPM_clientNotifyEnter(deepest_lpm);
120 
121  switch (deepest_lpm) {
122  case LOW_POWER_MODE_NONE:
123  mgr_lpm_ctxt->low_power_mode = deepest_lpm;
124  break;
125  case LOW_POWER_MODE_SLEEP:
126  mgr_lpm_ctxt->low_power_mode = deepest_lpm;
127  vMGR_LPM_enterSleep(env_config);
128  break;
129  case LOW_POWER_MODE_STOP:
130  mgr_lpm_ctxt->low_power_mode = deepest_lpm;
131  vMGR_LPM_enterStop(env_config);
132  break;
133  case LOW_POWER_MODE_STANDBY:
134  mgr_lpm_ctxt->low_power_mode = deepest_lpm;
135  vMGR_LPM_enterStandBy(env_config);
136  break;
137  case LOW_POWER_MODE_SHUTDOWN:
138  mgr_lpm_ctxt->low_power_mode = deepest_lpm;
139  vMGR_LPM_enterShutdown(env_config);
140  break;
141  default:
142  return KNS_STATUS_ERROR;
143  }
144 
145  //> Notify each client after exiting LPM
146  vMGR_LPM_clientNotifyExit(deepest_lpm);
147 
148  return KNS_STATUS_OK;
149 }

153 /**
154  * @brief internal API used to request client deepest LPM.
155  *
156  * @return The deepest LPM allowed by clients
157  */
158 static enum MgrLpm_LPM_t eMGR_LPM_clientRequest(void)
159 {
160  uint8_t index;
161 
162  enum MgrLpm_LPM_t client_deepest_lpm;
163  enum MgrLpm_LPM_t deepest_lpm;
164 
165  deepest_lpm = LOW_POWER_MODE_SHUTDOWN;
166 
167  if (mgrLpmClientNbr == 0)
168  return LOW_POWER_MODE_NONE;
169 
170  //> Request the deepest LPM allowed by clients
171  for (index = 0; index < MGR_LPM_CLIENT_NBR_MAX; index++) {
172  if (mgrLpmClientTab[index].fpMGR_LPM_LpmReqCb != NULL) {
173  client_deepest_lpm = mgrLpmClientTab[index].fpMGR_LPM_LpmReqCb();
174  if (client_deepest_lpm < deepest_lpm)
175  deepest_lpm = client_deepest_lpm;
176  }
177  }
178 
179  return deepest_lpm;
180 }

275 static void vMGR_LPM_enterStandBy(struct MgrLpm_EnvConfig_t env_config)
276 {
277  if (env_config.fp_standby_enter != NULL)
278  env_config.fp_standby_enter();
279 
280  /* Clear all wakeup flags in status register. done just before entering sleep
281  * to ensure rising edge on wakeup will be occur during the sleep period
282  */
283  __HAL_PWR_CLEAR_FLAG(PWR_FLAG_SB);
284  __HAL_PWR_CLEAR_FLAG(PWR_FLAG_WU);
285  __HAL_PWR_CLEAR_FLAG(PWR_FLAG_WUFI);
286  HAL_PWR_EnterSTANDBYMode();
287 
288  /* We should never reach this point as standby exit performs a reset of the uC */
289  NVIC_SystemReset();
290  kns_assert(0);
291 }
