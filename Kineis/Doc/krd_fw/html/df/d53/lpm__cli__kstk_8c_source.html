30 enum MgrLpm_LPM_t KSTK_lpmReq(void)
31 {
32  union KNS_MAC_rsrcStatus_t rsrc;
33 
34  rsrc = KNS_MAC_getRsrcStatus();
35 
36  /** On STM32WL55C + kineis stack, possible LPM strategy is:
37  * * RUNTIME when woken-up to:
38  * * configure/start the SUBGHZ RF through SPI bus
39  * * start the TX- TIMEOUT timer
40  * * LPSLEEP or STOP during SUBGHZ transmission, until TX DONE or until TX TIMEOUT
41  * * STANDBY in any other situation (RTC timer is used for periodical transmission
42  * * SHUTDOWN when there is no ressource to backup.
43  * @attention KNS-MAC-ressource status is an internal variable which acyually needs to
44  * remain over LPM, but when all is cleared, it is ok to enter a LPM (e.g. shutdown)
45  * which clears this variables (as it is already cleared!!)
46  *
47  * * @note SUBGHZ peripheral is able to wakeup the STM32WL55xx uC from STNDBY, but the timer
48  * TIM16 is only able to wakeup from LPSLEEP
49  * */
50  if (rsrc.raw == 0x0)
51  return LOW_POWER_MODE_SHUTDOWN;
52  if (rsrc.txTimeout == 1) // as TX timeout timer is htim16 (cf aks_l1_cfg.h)
53  return LOW_POWER_MODE_SLEEP;
54  return LOW_POWER_MODE_STANDBY;
55 }
