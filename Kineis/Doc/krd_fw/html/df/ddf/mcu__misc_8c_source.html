23 /** @note Define a delay macro which does not depend on systick interrupt in a way to be able to run
24  * from any ISR (e.g. turn ON PA wrapper below will be called from SUBGHZ ISR during continuous
25  * modulated wave)
26  *
27  * @attention at typical 32MHz clock and 4 instruction cycle per loop, max value is 536870 ms
28  *
29  * NOP : 1 cycle
30  * SUBS R3, #1 : 1 cycle
31  * CMP R3, #0 : 1 cycle
32  * BNE <ADDR> : 1 cycle
33  */
34 //#define DELAY_MS(time_ms) HAL_Delay(time_ms)
35 extern uint32_t SystemCoreClock;
36 #define FOR_LOOP_CYCLE_NB 4
37 #define DELAY_MS(time_ms) \
38 { \
39  uint32_t _nb_for_loop_per_ms = SystemCoreClock / 1000 / FOR_LOOP_CYCLE_NB; \
40  for (uint32_t _count = time_ms * _nb_for_loop_per_ms; \
41  _count > 0 ; \
42  _count--) __NOP(); \
43 }

48 {
49  /** @attention this code may run under ISR, especially during continuous modulated wave */
50 #ifdef KRD_FW_MP
51  GPIO_InitTypeDef GPIO_InitStruct = {0};
52 
53 
54  /* GPIO Ports Clock Enable */
55  __HAL_RCC_GPIOA_CLK_ENABLE();
56 
57  /*Configure GPIO pin Output Level */
58  HAL_GPIO_WritePin(PA_PSU_EN_GPIO_Port, PA_PSU_EN_Pin, GPIO_PIN_RESET);
59 
60 
61  /*Configure GPIO pin : PtPin */
62  GPIO_InitStruct.Pin = PA_PSU_EN_Pin;
63  GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
64  GPIO_InitStruct.Pull = GPIO_NOPULL;
65  GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
66  HAL_GPIO_Init(PA_PSU_EN_GPIO_Port, &GPIO_InitStruct);
67 
68 
69 
70  HAL_GPIO_WritePin(PA_PSU_EN_GPIO_Port, PA_PSU_EN_Pin, GPIO_PIN_SET);
71  DELAY_MS(10);
72 #endif
73 }

93 {

95  int8_t rficPaLevel, rficPaLevel_min, rficPaLevel_max;
96 
97

100  rficPaLevel_max = 15; // dBm
101 
102  settings->isRficHPA = false;
103  settings->externalPaGain = 36;
104 #elif defined(KRD_FW_LP)
105  rficPaLevel_min = -9; // dBm
106  rficPaLevel_max = 22; // dBm
107 
108  settings->isRficHPA = true;
109  settings->externalPaGain = 0;
110 #endif
111  rficPaLevel = rf_level_dBm - settings->externalPaGain;
112 
113  MGR_LOG_VERBOSE("[MCU_MISC] RFIC PA select (0:LPA, 1:HPA) = %d\r\n",
114  settings->isRficHPA);
115  MGR_LOG_VERBOSE("[MCU_MISC] RFIC PA level=%d dBm, external PA gain = %d dB\r\n",
116  rficPaLevel,
117  settings->externalPaGain);
118 
119  /* check settings */
120  if ((rficPaLevel >= rficPaLevel_min) && (rficPaLevel <= rficPaLevel_max))
121  settings->rficPaLevel = rficPaLevel;
122  else
123  return KNS_STATUS_BAD_SETTING;
124 
125  return KNS_STATUS_OK;
