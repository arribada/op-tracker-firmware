34 bool bMGR_AT_CMD_sendResponse(enum atcmd_rsp_type_t atcmd_response_type, void *atcmd_rsp_data)
35 {
36 
37  switch (atcmd_response_type) {
38  case ATCMD_RSP_TXOK:
39  {
40  if (atcmd_rsp_data != NULL) {
41  struct sUserDataTxFifoElt_t *spUserDataMsg =
42  (struct sUserDataTxFifoElt_t *)atcmd_rsp_data;
43  uint8_t *pu8UserDataPtr = spUserDataMsg->u8DataBuf;
44  uint16_t u16UserDataBitlen = spUserDataMsg->u16DataBitLen;
45 
46  MCU_AT_CONSOLE_send("+TX=0,");
47  MCU_AT_CONSOLE_send_dataBuf(pu8UserDataPtr, u16UserDataBitlen);
48  MCU_AT_CONSOLE_send("\r\n");
49  }
50  return true;
51  }
52  break;
53  case ATCMD_RSP_RXTIMEOUT:
54  case ATCMD_RSP_TXTIMEOUT:
55  case ATCMD_RSP_RXERROR:
56  {
57  if (atcmd_rsp_data != NULL) {
58  struct sUserDataTxFifoElt_t *spUserDataMsg =
59  (struct sUserDataTxFifoElt_t *)atcmd_rsp_data;
60  uint8_t *pu8UserDataPtr = spUserDataMsg->u8DataBuf;
61  uint16_t u16UserDataBitlen = spUserDataMsg->u16DataBitLen;
62  enum ERROR_RETURN_T error_id;
63 
64  if (atcmd_response_type == ATCMD_RSP_RXERROR)
65  error_id = (enum ERROR_RETURN_T) KNS_STATUS_RF_ERR;
66  else
67  error_id = (enum ERROR_RETURN_T) KNS_STATUS_TIMEOUT;
68 
69  MCU_AT_CONSOLE_send("+TX=%d,", error_id);
70  MCU_AT_CONSOLE_send_dataBuf(pu8UserDataPtr, u16UserDataBitlen);
71  MCU_AT_CONSOLE_send("\r\n");
72  }
73  return true;
74  }
75  break;
76  case ATCMD_RSP_TXACKOK:
77  {
78  MCU_AT_CONSOLE_send("+TXACK=0\r\n");
79  return true;
80  }
81  break;
82  case ATCMD_RSP_TXACKNOTOK:
83  {
84  enum ERROR_RETURN_T error_id = (enum ERROR_RETURN_T) KNS_STATUS_TIMEOUT;
85 
86  MCU_AT_CONSOLE_send("+TXACK=%d\r\n", error_id);
87  return true;
88  }
89  break;
90  case ATCMD_RSP_SATDET:
91  {
92  if (atcmd_rsp_data != NULL) {
93  struct KNS_MAC_SATDET_ctxt_t *det_info =
94  (struct KNS_MAC_SATDET_ctxt_t *)atcmd_rsp_data;
95  uint32_t detected_freq = det_info->detected_freq;
96  uint32_t detect_duration = det_info->detect_duration;
97  float rssi = det_info->rssi;
98 
99  MCU_AT_CONSOLE_send("+SATDET=%ld,%ld,%f\r\n", detect_duration,
100  detected_freq, rssi);
101  }
102  return true;
103  }
104  break;
105  case ATCMD_RSP_SATLOST:
106  {
107  MCU_AT_CONSOLE_send("+SATLOST=\r\n");
108  return true;
109  }
110  break;
111  case ATCMD_RSP_SATDETTO:
112  {
113  MCU_AT_CONSOLE_send("+SATDET=0,0,0\r\n");
114  return true;
115  }
116  break;
117  case ATCMD_RSP_RXOK:
118  {
119  if (atcmd_rsp_data != NULL) {
120  struct KNS_MAC_RX_frm_ctxt_t *receivedFrm =
121  (struct KNS_MAC_RX_frm_ctxt_t *)atcmd_rsp_data;
122  uint8_t *rxFrmDataPtr = receivedFrm->data;
123  uint16_t rxFrmDataBitlen = receivedFrm->data_bitlen;
124  float rssi = receivedFrm->rssi;
125 
126  MCU_AT_CONSOLE_send("+RX=");
127  MCU_AT_CONSOLE_send_dataBuf(rxFrmDataPtr, rxFrmDataBitlen);
128  MCU_AT_CONSOLE_send(",%f\r\n", rssi);
129  }
130  return true;
131  }
132  break;
133  case ATCMD_RSP_DLOK:
134  {
135  if (atcmd_rsp_data != NULL) {
136  struct KNS_MAC_RX_frm_ctxt_t *reeceivedMsg =
137  (struct KNS_MAC_RX_frm_ctxt_t *)atcmd_rsp_data;
138  uint8_t *rxMsgDataPtr = reeceivedMsg->data;
139  uint16_t rxMsgDataBitlen = reeceivedMsg->data_bitlen;
140 
141  MCU_AT_CONSOLE_send("+DL=");
142  MCU_AT_CONSOLE_send_dataBuf(rxMsgDataPtr, rxMsgDataBitlen);
143  MCU_AT_CONSOLE_send("\r\n");
144  }
145  return true;
146  }
147  break;
148  default:
149  break;
150  }
151  return false;
152 }
